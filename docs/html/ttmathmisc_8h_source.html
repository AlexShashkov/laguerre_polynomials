181static uint CharToDigit(uint c)
182{
183 if(c>='0' && c<='9')
184 return c-'0';
185
186 if(c>='a' && c<='z')
187 return c-'a'+10;
188
189return c-'A'+10;
190}

202static sint CharToDigit(uint c, uint base)
203{
204 if( c>='0' && c<='9' )
205 c=c-'0';
206 else
207 if( c>='a' && c<='z' )
208 c=c-'a'+10;
209 else
210 if( c>='A' && c<='Z' )
211 c=c-'A'+10;
212 else
213 return -1;
214
215
216 if( c >= base )
217 return -1;
218
219
220return sint(c);
221}
